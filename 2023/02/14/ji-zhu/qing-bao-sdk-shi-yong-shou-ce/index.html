class SDK():

    def __init__(self) -> None:
        self.p = None
        self.soObj = None

    def initSDK(self):
        dllabspath = os.path.dirname(os.path.abspath(__file__)) + os.path.sep + '../../lib/libsdk.so'
        configabspath = os.path.dirname(os.path.abspath(__file__)) + os.path.sep + '../../conf/config.properties'
        
        so = ctypes.CDLL(dllabspath)

        so.query.restype = ctypes.c_char_p
        so.add.restype = ctypes.c_char_p
        so.modify.restype = ctypes.c_char_p
        so.queryHash.restype = ctypes.c_char_p
        so.queryCreditDetails.restype = ctypes.c_char_p
        
        # еҲқе§ӢеҢ–sdk
        ptr = so.initSDK(ctypes.c_char_p(configabspath.encode("utf-8")))    
        self.p = ctypes.c_void_p(ptr)
        self.soObj = so

    def query(self, type, key):
        res = self.soObj.query(self.p, ctypes.c_char_p(type.encode("utf-8")), ctypes.c_char_p(key.encode("utf-8")))
        return res.decode()
    
    def add(self, type, key, value):
        t = ctypes.c_char_p(type.encode("utf-8"))
        k = ctypes.c_char_p(key.encode("utf-8"))
        v = ctypes.c_char_p(value.encode("utf-8"))
        res = self.soObj.add(self.p, t, k, v)
        return res.decode()

    def modify(self, type, key, value):
        t = ctypes.c_char_p(type.encode("utf-8"))
        k = ctypes.c_char_p(key.encode("utf-8"))
        v = ctypes.c_char_p(value.encode("utf-8"))
        res = self.soObj.modify(self.p, t, k, v)
        return res.decode()

    def delete(self, type, key):
        t = ctypes.c_char_p(type.encode("utf-8"))
        k = ctypes.c_char_p(key.encode("utf-8"))
        self.soObj.delete(self.p, t, k)
    
    def queryHash(self):
        res = self.soObj.queryHash(self.p)
        return res.decode()

    def queryCreditDetails(self):
        res = self.soObj.queryCreditDetails(self.p)
        return res.decode()

    def destroy(self):
        if self.p == 0:
            return
        self.soObj.destroy(self.p)

еҪўејҸдёәLinuxдёӢзҡ„еҸҜжү§иЎҢж–Үд»¶гҖӮзӣҙжҺҘиҝҗиЎҢжӯӨж–Үд»¶пјҢеҚіеҸҜеңЁCLIдёӯеҜ№жғ…жҠҘж•°жҚ®иҝӣиЎҢж“ҚдҪңпјҢеҜ№жңәеҷЁhashеҸҠиҜҒд№ҰиҜҰжғ…иҝӣиЎҢжҹҘиҜўгҖӮ

жү§иЎҢcliпјҢжҹҘзңӢе…¶еҢ…еҗ«зҡ„жүҖжңүе‘Ҫд»ӨеҸҠи§ЈйҮҠиҜҙжҳҺгҖӮ

е‘Ҫд»Ө иҜҙжҳҺ
-g, вҖ“config еҜје…Ҙй…ҚзҪ®ж–Үд»¶пјҢеҰӮжһңдёҚеҶҷпјҢеҲҷй»ҳи®ӨеңЁconfзӣ®еҪ•пјҲзӣ®еүҚзҡ„зӣёеҜ№дҪҚзҪ®пјүдёӯжү«жҸҸй…ҚзҪ®
-f, вҖ“filePath еҜје…ҘеҠ еҜҶж•°жҚ®иҮіеҶ…еӯҳжҲ–redisеӯҳеӮЁ
-t, вҖ“operateType иҰҒж“ҚдҪңзҡ„ж•°жҚ®зұ»еһӢпјҢжңүipгҖҒdomainгҖҒsimpleгҖҒurlеӣӣз§Қ
-a, вҖ“add еўһеҠ ж•°жҚ®пјҢиҫ“е…Ҙж јејҸдёәkeyе’Ңvalueз»„жҲҗзҡ„jsonз»“жһ„дҪ“пјҢиҜҰжғ…еҸӮи§Ғжғ…жҠҘж•°жҚ®з»“жһ„дҪ“иҜҙжҳҺ
-d, вҖ“delete еҲ йҷӨж•°жҚ®пјҢеҰӮжһңиҰҒеҲ йҷӨеӨҡжқЎж•°жҚ®пјҢkeyд№Ӣй—ҙд»Ҙз©әж јдҪңдёәеҲҶйҡ”з¬ҰгҖӮ
-q, вҖ“query жҹҘиҜўж•°жҚ®пјҢеҰӮжһңиҰҒжҹҘиҜўеӨҡжқЎж•°жҚ®пјҢkeyд№Ӣй—ҙд»Ҙз©әж јдҪңдёәеҲҶйҡ”з¬ҰгҖӮ
-m, вҖ“modify дҝ®ж”№ж•°жҚ®пјҢиҫ“е…Ҙж јејҸдёәkeyе’Ңvalueз»„жҲҗзҡ„jsonз»“жһ„дҪ“пјҢиҜҰжғ…еҸӮи§Ғжғ…жҠҘж•°жҚ®з»“жһ„дҪ“иҜҙжҳҺ
вҖ“hash жҹҘиҜўи®ҫеӨҮhash
вҖ“certificate жҹҘиҜўиҜҒд№ҰиҜҰжғ…

  1. йңҖиҰҒе°ҶеҠЁжҖҒй“ҫжҺҘеә“ж–Үд»¶еҠ иҪҪиҝӣзі»з»ҹзҺҜеўғеҸҳйҮҸжүҚиғҪжӯЈеёёдҪҝз”ЁSDKгҖӮе…·дҪ“жӯҘйӘӨеҰӮдёӢгҖӮ

    ж–№жЎҲ1пјҡдёҙж—¶ж–№жЎҲгҖӮ

    еҒҮеҰӮsoж–Үд»¶еңЁ/root/goProject/src/xx-SDK/cдёӢ, еҲҷеңЁcmdдёӯжү§иЎҢ:

    ж–№жЎҲ2: ж°ёд№…ж–№жЎҲгҖӮ

    LinuxеҠЁжҖҒеә“зҡ„й»ҳи®Өжҗңзҙўи·Ҝеҫ„жҳҜ/libе’Ң/usr/libпјҢжҲ‘д»¬еҸҜд»Ҙе°Ҷsoж–Үд»¶ж”ҫеҲ°иҝҷдёӨдёӘзӣ®еҪ•дёӯпјҢжҲ–иҖ…з”Ёд»ҘдёӢж–№ејҸжҢҮе®ҡжү«жҸҸи·Ҝеҫ„гҖӮдёӨз§Қж–№ејҸйғҪиғҪж°ёд№…з”ҹж•ҲгҖӮ

    зј–иҫ‘etcдёӢзҡ„ld.so.conf

    vim /etc/ld.so.conf
